   render() {
     const showExpandHint = this.contentMode === 'expandable' && !this._isExpanded && this.contentText.length > 0;
     const contentClasses = {
       content: true,
       collapsed: this.contentMode === 'expandable' && !this._isExpanded,
       expanded: this.contentMode === 'expandable' && this._isExpanded,
       scrollable: this.contentMode === 'scrollable'
     };
     const iconClasses = { icon: true, expanded: this._isExpanded };
     const totalTokens = this.inputTokens !== undefined && this.outputTokens !== undefined
       ? this.inputTokens + this.outputTokens
       : undefined;

     return html`
       <div class="card">
         <div class="header" @click="${this.toggleExpanded}">
           <div class="${classMap(iconClasses)}">
             ${this.contentMode === 'expandable' ? this._renderChevronIcon() : this._renderToolIcon()}
           </div>
           <div class="tool-info">
             <div class="tool-name">
               ${this.toolName}
               ${this._isStreaming ? html`<span class="streaming-indicator"></span>` : nothing}
             </div>
             <div class="tool-args">${this._formatArgs(this.toolArguments)}</div>
           </div>
           <div class="meta">
             ${totalTokens !== undefined ? html`
               <div class="meta-item" title="${this.inputTokens} in / ${this.outputTokens} out">
                 ${this._renderTokenIcon()}
                 <span>${totalTokens.toLocaleString()}</span>
               </div>
             ` : nothing}
             ${this.durationMs !== undefined ? html`
               <span class="duration-badge">${this._formatDuration(this.durationMs)}</span>
             ` : nothing}
           </div>
         </div>
         ${this.contentText ? html`
           <div class="content-wrapper">
             <div class="${classMap(contentClasses)}">
               ${this.contentText}
             </div>
           </div>
           ${showExpandHint ? html`
             <div class="expand-hint" @click="${this.toggleExpanded}">
               Click to expand (${this.contentText.split('\n').length} lines)
             </div>
           ` : nothing}
         ` : html`
           <div class="content empty-state">Waiting for output...</div>
         `}
       </div>
     `;
   }
 }
